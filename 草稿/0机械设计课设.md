$$
\begin{align}
P_{w}&=\dfrac{Fv}{1000}=F \cdot\dfrac{n}{60}\cdot2\pi\cdot  \dfrac{D}{2}\cdot10^{-3}\cdot 10^{-3}=4.189 kW \\
\eta_{a}&=\eta_{1}\eta_{2}\eta_{3}\eta_{4}\eta_{5}\eta_{6}\eta_{7}=85.91\%
\end{align}
$$

$$

\begin{align}
